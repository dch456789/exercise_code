#include "0429.2_1.h"
using namespace std;

class point
{
private:
	int x;
	int y;
public:
	void init(void)
	{
		this->x = 0;
		this->y = 0;
	}
	void init(int x)
	{
		this->x = x;
		this->y = x;
	}
	void init(int x, int y)
	{
		this->x = x;
		this->y = y;
	}
	void init(const point& p)
	{
		this->x = p.x;
		this->y = p.y;
	}
	void toprint(void)
	{
		std::cout << "Point(" << x << "," << y << ")";
	}
	int& getX(void)
	{
		return this->x;
	}
	int& getY(void)
	{
		return this->y;
	}
};
class circle
{
private:
	point p;
	int r;
public:
	void init(int x = 0, int y = 0, int r = 0)
	{
		this->p.init(x, y);
		this->r = r;
	}
	void toprint(void)
	{
		std::cout << "Circle:([x=" << this->p.getX() << "][y=" << this->p.getY() << "][r=" << this->r << "])";
	}
	int& getX(void)
	{
		return this->p.getX();
	}
	int& getY(void)
	{
		return this->p.getY();
	}
	int& getR(void)
	{
		return this->r;
	}
};
